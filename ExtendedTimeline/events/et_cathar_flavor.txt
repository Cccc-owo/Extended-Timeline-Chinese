namespace = et_cathar

country_event = {
    #Start Cathar Reformation
    id = et_cathar.1
    title = "et_cathar.EVTNAME1"
    desc = "et_cathar.EVTDESC1"
    picture = REFORM_eventPicture


    trigger = {
		religion = catholic
		NOT = { is_religion_enabled = cathar }
       	reform_desire = 1
		capital_scope = { continent = europe }
		NOT = { tag = PAP }
	}

	major = yes

	fire_only_once = yes

	mean_time_to_happen = {
        months = 240

		modifier = {
			factor = 0.75
			NOT = { num_of_cities = 10 }
		}
		modifier = {
			factor = 0.75
			NOT = { num_of_cities = 2 }
		}
		modifier = {
			factor = 0.95
			NOT = { has_opinion = { who = PAP value = -50 } }
		}
		modifier = {
			factor = 0.85
			NOT = { has_opinion = { who = PAP value = -100 } }
		}
		modifier = {
			factor = 0.75
			NOT = { has_opinion = { who = PAP value = -150 } }
		}
		modifier = {
			factor = 2.0
			government = theocracy
		}
		modifier = {
			factor = 0.95
			any_owned_province = {
				religion = bogomilism
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_global_flag = cathar_reformation_started_et
			add_reform_desire = -1
			random_owned_province = {
				save_event_target_as = target_province
			}
			random_owned_province = {
				limit = {
					religion = bogomilism
				}
				save_event_target_as = target_province
			}
		}
	}

	option = {
		name = "et_cathar.1.a"
		enable_religion = cathar
		add_stability = -1
		event_target:target_province = {
			change_religion = cathar
			add_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
			add_province_modifier = {
				name = "cathar_persecution"
				duration = 9000
			}
			if = {
				limit = {
					NOT = { is_capital = yes }
				}
				root = {
					capital_scope = {
						change_religion = cathar
						add_permanent_province_modifier = {
							name = "religious_zeal_at_conv"
							duration = 9000
						}
						add_province_modifier = {
							name = "cathar_persecution"
							duration = 9000
						}
					}
				}
			}
		}
	}

	option = {
		name = "et_cathar.1.b"
		enable_religion = cathar
		hidden_effect = {
			set_country_flag = tolerance_of_cathars
		}
		add_papal_influence = -50
		event_target:target_province = {
			change_religion = cathar
			add_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
			if = {
				limit = {
					NOT = { is_capital = yes }
				}
				root = {
					capital_scope = {
						change_religion = cathar
						add_permanent_province_modifier = {
							name = "religious_zeal_at_conv"
							duration = 9000
						}
					}
				}
			}
		}
	}
}

country_event = {
    #Pope calls for Cathar Crusade
    id = et_cathar.4
    title = "et_cathar.EVTNAME4"
    desc = "et_cathar.EVTDESC4"
    picture = POPE_PREACHING_eventPicture

    major = yes
	fire_only_once = yes

    trigger = {
		religion = catholic
       	OR = {
			tag = PAP
			AND = {
				NOT = { exists = PAP }
				owns = 118
			}
		}
		any_country = {
			religion = cathar
		}
	}
	mean_time_to_happen = {
        months = 120

	}
	option = { #Call for Crusade
		name = "et_cathar.4.a"

		every_country = {
			limit = { religion = catholic }
			every_country = {
				limit = { religion = cathar }
				reverse_add_casus_belli = {
					target = PREV
					type = cb_heretic
					months = 240
				}
			}
			country_event = {
				id = et_cathar.5
			}
		}

		ai_chance = {
			factor = 99
		}
	}
	option = {	# Don't go for crusade
		name = "et_cathar.4.b"

		add_prestige = -10

		ai_chance = {
			factor = 1
		}

	}
}
country_event = {
    #Support or deny crusade
    id = et_cathar.5
    title = "et_cathar.EVTNAME5"
    desc = "et_cathar.EVTDESC5"
    picture = RELIGIOUS_TURMOIL_eventPicture

	is_triggered_only = yes

	option = { #Support the Crusade
		name = "et_cathar.5.a"

		if = {
			limit = {
				exists = PAP
			}
			PAP = {
				add_opinion = {
					who = ROOT
					modifier = supports_cathar_crusade
				}
			}
		}
		every_country = {
			limit = { religion = cathar }
			add_opinion = {
				who = ROOT
				modifier = supports_cathar_crusade_cathar
			}
		}

		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.1
				exists = PAP
				NOT = {
					has_opinion = {
						who = PAP
						value = 0
					}
				}
			}
			modifier = {
				factor = 2
				exists = PAP
				has_opinion = {
					who = PAP
					value = 50
				}
			}
			modifier = {
				factor = 0.1
				any_ally = { religion = cathar }
			}
			modifier = {
				factor = 10
				any_rival_country = { religion = cathar }
			}
		}
	}
	option = {	#Refuse to help
		name = "et_cathar.5.b"

		if = {
			limit = {
				exists = PAP
			}
			PAP = {
				add_opinion = {
					who = ROOT
					modifier = no_support_cathar_crusade
				}
			}
		}

		ai_chance = {
			factor = 50
		}
	}
}
