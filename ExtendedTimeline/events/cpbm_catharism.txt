# Pelican spotted

province_event = {
	id = cpbm_cathar.4
	title = cpbm_cathar.EVTNAME4
	desc = cpbm_cathar.EVTDESC4
	picture = RELIGION_eventPicture

	trigger = {
		religion = cathar
		owner = { religion = cathar }
		NOT = { has_climate = arid }
		NOT = { has_province_flag = pelican_spotted }
	}

	mean_time_to_happen = {
		months = 2500
	}

	immediate = {
		hidden_effect = {
			set_province_flag = pelican_spotted
		}
	}

	option = {
		name = cpbm_cathar.EVTOPT4
		owner = { add_stability = 1 }
	}
}

				}

# The Troubadours

province_event = {
	id = cpbm_cathar.16
	title = cpbm_cathar.EVTNAME16
	desc = cpbm_cathar.EVTDESC16
	picture = CULTURE_eventPicture
	
	fire_only_once = yes

	trigger = {
		culture = occitain
		is_year = 1096
	}

	mean_time_to_happen = {
		months = 842

		modifier = {
			factor = 0.6
			owner = { religion = cathar }
		}

		modifier = {
			factor = 0.85
			region = france_region
		}
	}

	option = {
		name = cpbm_cathar.EVTOPT16
		add_province_modifier = {
			name = troubadours
			duration = -1
		}
	}
}
