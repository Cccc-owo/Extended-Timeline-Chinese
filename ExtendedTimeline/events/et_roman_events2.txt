			owner = { has_reform = barbarian_reform }

		months = 5000

			factor = 1.5

		}

			owner = { stability = 2 }
		}
		modifier = {
			factor = 0.75
			owner = { NOT = { stability = 0 } }
		}
		modifier = {
			factor = 0.75
			owner = { NOT = { stability = -1 } }
		}
		modifier = {
			factor = 0.75
			owner = { NOT = { stability = -2 } }
		}
	}

	option = {
		name = "nov.EVTOPTA20"
		owner = { add_prestige = -10 }

		}
	}
