#African colonial culture change
country_event = {
	id = culture.9
	title = culture.9.t
	desc = culture.9.d
	picture = STREET_SPEECH_eventPicture

	trigger = {
		is_former_colonial_nation = yes
		is_subject = no
		capital_scope = { 
			continent = africa
			NOT = { culture = ROOT } 
		}
		NOT = { has_country_flag = african_culture_change_event }
		NOT = { culture_group = southern_african }
		NOT = { culture_group = kongo_group }
		NOT = { culture_group = great_lakes_group }
		NOT = { culture_group = african }
		NOT = { culture_group = west_african }
		NOT = { culture_group = sahelian }
		NOT = { culture_group = mande }
	}

	mean_time_to_happen = {
		months = 6
	}
	
	immediate = {
		set_country_flag = african_culture_change_event
	}

	option = {
		name = culture.9.a
		change_primary_culture = capital
		ai_chance = { 
			factor = 100
		}
	}
	option = {
		name = culture.9.b
		add_stability = -1
		ai_chance = { 
			factor = 0
		}
	}
}

		has_country_modifier = cultural_assimilation_enforced
		any_owned_province = {
			has_owner_culture = no
		}
	}

	mean_time_to_happen = {
		months = 40
		modifier = {
			factor = 0.75
			has_reform = fascist_government
		}
		modifier = {
			factor = 0.75
			autocracy_ideas = 6
		}
	}

	option = {
		name = culture.12.a
		if = {
			limit = {
				any_owned_province = {
					has_owner_culture = no
					has_owner_accepted_culture = no
					any_neighbor_province = {
						has_owner_culture = yes
					}
				}
			}
			random_owned_province = {
				limit = {
					has_owner_culture = no
					has_owner_accepted_culture = no
					any_neighbor_province = {
						has_owner_culture = yes
					}
				}
				change_culture = ROOT
			}
		}
		else_if = {
			limit = {
				any_owned_province = {
					has_owner_culture = no
					has_owner_accepted_culture = no
				}
			}
			random_owned_province = {
				limit = {
					has_owner_culture = no
					has_owner_accepted_culture = no
				}
				change_culture = ROOT
			}
		}
		else = {
			random_owned_province = {
				limit = {
					has_owner_culture = no
				}
				change_culture = ROOT
			}
		}
	}
