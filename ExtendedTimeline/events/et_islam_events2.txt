#Sunni-Shia Split - Umayyads
country_event = {
	id = et_islam.13
	title = "et_islam.13.t"
	desc = "et_islam.13.d"
	picture = CIVIL_WAR_eventPicture
	
	is_triggered_only = yes
	fire_only_once = yes
	
	option = {
		name = "et_islam.13.a"
		change_government = monarchy
		add_government_reform = caliphate_reform
		define_heir = {
			name = "Yazid"
			dynasty = "Umayyad"
			claim = 100
			age = 14
			adm = 4
			dip = 5
			mil = 5
		}
		if = {
			limit = { NOT = { is_religion_enabled = shiite } }
			enable_religion = shiite
		}
		if = {
			limit = { NOT = { is_religion_enabled = sunni } }
			enable_religion = sunni
		}
		change_religion = sunni
		if = {
			limit = { iraq_arabi_area = { owned_by = ROOT } }
			random_owned_province = {
				limit = { area = iraq_arabi_area }
				change_religion = shiite
				shiite_rebels = 4
			}
		}
		if = {
			limit = { basra_area = { owned_by = ROOT } }
			random_owned_province = {
				limit = { area = basra_area }
				change_religion = shiite
				shiite_rebels = 4
			}
		}
		if = {
			limit = { bahrain_area = { owned_by = ROOT } }
			random_owned_province = {
				limit = { area = bahrain_area }
				change_religion = shiite
				shiite_rebels = 4
			}
		}
		random_owned_province = {
			change_religion = shiite
			shiite_rebels = 4
		}
		every_province = {
			limit = { religion = islam }
			change_religion = sunni
		}
		every_country = {
			limit = { religion = islam }
			change_religion = sunni
		}
		if = {
			limit = { has_country_modifier = rise_of_islam }
			remove_country_modifier = rise_of_islam
		}
		hidden_effect = {
			clr_country_flag = first_fitna_can_not_end
		}
	}
}
