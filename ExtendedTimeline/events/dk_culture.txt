#Birth of Catalan Culture
province_event = {
	id = culturedk.9
	title = culturedk.1.t
	desc = culturedk.1.d
	picture = CULTURE_eventPicture
	trigger = {
		NOT = { is_year = 1000 }
		is_year = 550
		OR = {
			culture = celtiberian
			culture = lusitanian
			culture = turduli
			culture = cantabrian
			culture = gallaecian
			culture = iberi
			culture = roman
		}
		OR = {
			area = catalonia_area
			area = valencia_area
		}
		owner = {
			NOT = { primary_culture = roman }
			NOT = { primary_culture = iberi }
			NOT = { primary_culture = celtiberian }
			NOT = { primary_culture = lusitanian }
			NOT = { primary_culture = turduli }
			NOT = { primary_culture = cantabrian }
			NOT = { primary_culture = gallaecian }
			NOT = { culture_group = basque_group }
		}
	}
	mean_time_to_happen = {
		years = 50
	}
	option = {
		name = culturedk.1.a
		change_culture = catalan
	}
}
