}

#The Fate of Texas
country_event = {
	id = tex.5
	title = tex.5.t
	desc = tex.5.d
	picture = COLONIZATION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = TEX
		is_at_war = no	
		is_year = 1845
		ai = yes
		exists = USA
		government = republic
		USA = { primary_culture = ROOT }
	}
	
	mean_time_to_happen = {
		months = 10
	}
	
	option = {
		name = "tex.5.a"
		USA = {
			country_event = { id = tex.6 }
		}
	}
	option = {
		name = "tex.5.b"
		california_region = { 
			add_permanent_claim = TEX
		} 
		add_stability = -1
		add_country_modifier = {
			name = tex_west
			duration = 18250
		}
	}
}
