#Asking the Oracles
country_event = {
	id = egyptian.4
	title = egyptian.4.t
	desc = egyptian.4.d
	picture = PRAYING_eventPicture

	trigger = {
		religion = egyptian
		NOT = { has_ruler_modifier = good_fortune }
		NOT = { has_ruler_modifier = bad_fortune }
	}

	mean_time_to_happen = {
		months = 600
	}

	option = {
		name = egyptian.4.a
		hidden_effect = {
			random_list = {
				10 = {
					country_event = {
						id = egyptian.5
						days = 10
					}
				}
				10 = {
					country_event = {
						id = egyptian.6
						days = 10
					}
				}
				80 = {
					country_event = {
						id = egyptian.7
						days = 10
					}
				}
			}
		}
	}

	option = {
		name = egyptian.4.b
		add_prestige = -2
	}
}
