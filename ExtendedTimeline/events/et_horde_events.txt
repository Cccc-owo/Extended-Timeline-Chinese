					owner = { tag = ROM }

			}
		}
	}

#AI horde DOW
country_event = {
	id = et.95
	title = "et.EVTNAME95"
	desc = "et.EVTDESC95"
	picture = HORDE_ON_HORSEBACK_eventPicture

	trigger = {
		has_country_modifier = nomadic_conquest
		is_at_war = no
		NOT = { war_exhaustion = 10 }
		manpower_percentage = 0.25
		ai = yes
		any_neighbor_country = { 
			NOT = { truce_with = ROOT } 
			NOT = { alliance_with = ROOT }
			NOT = { is_subject_of = ROOT }
			NOT = { overlord_of = ROOT }
			OR = {
				is_subject = no
				overlord = { 
					NOT = { overlord_of = ROOT } 
					NOT = { truce_with = ROOT } 
					NOT = { alliance_with = ROOT }
					NOT = { is_subject_of = ROOT }
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		name = "et.EVTOPTA63"
		add_stability = 1
		random_neighbor_country = {
			limit = { 
				NOT = { truce_with = ROOT } 
				NOT = { alliance_with = ROOT }
				NOT = { is_subject_of = ROOT }
			}
			ROOT = {
				declare_war_with_cb = {
					who = PREV
					casus_belli = cb_horde_vs_civ
				}
			}
		}
	}
}
