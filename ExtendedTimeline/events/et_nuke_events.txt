country_event = {
	id = et_nuke.2
	title = et_nuke.2.t
	desc = et_nuke.2.d
	picture = NUCLEAR_ENERGY_eventPicture

	is_triggered_only = yes
	
	option = { #Push the Big Red Button
		name = et_nuke.2.a
		change_variable = {
			which = num_of_nuclear_weapons
			value = -1
		}
		if = {
			limit = { NOT = { has_global_flag = nukes_used } }
			set_global_flag = nukes_used
			add_country_modifier = {
				name = used_nukes
				duration = 7300
			}
		}
		event_target:nuke_target_province = {
			province_event = {
				id = et_nuke.3
			}
			owner = {
				add_opinion = {
					who = ROOT
					modifier = nuked_us
				}
			}
		}
		clear_global_event_target = nuke_target_province
	}
	option = { #Abort Mission
		name = et_nuke.2.b
		clear_global_event_target = nuke_target_province
	}
}

province_event = {
	id = et_nuke.3
	title = et_nuke.3.t
	desc = et_nuke.3.d
	picture = NUCLEAR_ENERGY_eventPicture
	goto = ROOT

	is_triggered_only = yes
	
	option = { #
		name = et_nuke.3.a
		add_devastation = 100
		owner = { add_war_exhaustion = 4 }
		kill_units = { amount = 10 }
		add_province_modifier = {
			name = province_nuked
			duration = 1825
		}
		if = {
			limit = { base_tax = 20 }
			add_base_tax = -10
		}
		else_if = {
			limit = { base_tax = 6 }
			add_base_tax = -5
		}
		else = { set_base_tax = 1 }
		if = {
			limit = { base_production = 20 }
			add_base_production = -10
		}
		else_if = {
			limit = { base_production = 6 }
			add_base_production = -5
		}
		else = { set_base_production = 1 }
		if = {
			limit = { base_manpower = 20 }
			add_base_manpower = -10
		}
		else_if = {
			limit = { base_manpower = 6 }
			add_base_manpower = -5
		}
		else = { set_base_manpower = 1 }
	}
}
