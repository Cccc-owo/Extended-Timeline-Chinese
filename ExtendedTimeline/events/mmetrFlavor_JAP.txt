			}

			}

			remove_country_modifier = third_rank_court_rank_japan_mod
		}
		else_if = {
			limit = {
				has_ruler_flag = given_second_rank_court_rank_japan
				NOT = { has_ruler_flag = given_first_rank_court_rank_japan }
			}

				name = "first_rank_court_rank_japan_mod"
				duration = -1
			}
