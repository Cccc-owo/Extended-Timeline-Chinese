		required_missions = { koj_prepare_for_the_offensive_mission }
		position = 4
		provinces_to_highlight = {
			OR = {
				province_id = 379	#Jerusalem
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}

		trigger = {
			owns_core_province = 379

		}
		effect = {
			palestine_area = {
				limit = {
					NOT = { is_core = ROOT }

				}
				add_permanent_claim = ROOT
			}
		}
	}
