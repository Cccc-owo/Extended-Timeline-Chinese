  OR = {
    any_neighbor_province = {
      owner = {
        OR = {
          is_rival = root
          root = {
            is_rival = prev
          }
          AND = {
            is_subject_other_than_tributary_trigger = yes
            overlord = {
              OR = {
                is_rival = root
                root = {
                  is_rival = prev
                }
              }
            }
          }
        }
      }
    }
    any_neighbor_province = {
      owned_by = root
      any_neighbor_province = {
        owner = {
          OR = {
            is_rival = root
            root = {
              is_rival = prev
            }
            AND = {
              is_subject_other_than_tributary_trigger = yes

                OR = {
                  is_rival = root
                  root = {
                    is_rival = prev
